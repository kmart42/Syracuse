\begin{claim}
There is an integer integer larger than 10 that is not even.
\end{claim}
 
\begin{proof}
  Let $n$ = 11. \(11>10\) and \(n\nmid 2\). Therefore, a number n exists that is both greater than 10 and not even.
\end{proof}
